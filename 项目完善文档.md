**重连流程**:
```javascript
async function handleReconnect(playerId, sessionId) {
  // 1. 验证会话
  const session = await validateSession(sessionId);
  if (!session) return { success: false, reason: 'INVALID_SESSION' };
  
  // 2. 查找房间
  const room = await GameStore.findRoomByPlayer(playerId);
  if (!room) return { success: false, reason: 'ROOM_NOT_FOUND' };
  
  // 3. 发送状态快照
  const snapshot = room.game.getSnapshot();
  
  // 4. 回放最近 10 秒操作日志
  const recentLogs = await getOperationLogs(room.id, Date.now() - 10000);
  
  return {
    success: true,
    snapshot,
    recentLogs,
    currentPlayer: room.game.currentPlayer
  };
}
```

### 4.2 快照与回滚

#### 4.2.1 状态环缓冲区

```javascript
class StateHistory {
  constructor(maxSnapshots = 50) {
    this.snapshots = [];
    this.maxSnapshots = maxSnapshots;
    this.currentIndex = 0;
  }
  
  saveSnapshot(gameState, frameNumber) {
    const snapshot = {
      frame: frameNumber,
      timestamp: Date.now(),
      state: deepClone(gameState),
      checksum: calculateChecksum(gameState)
    };
    
    // 环形缓冲区：覆盖最旧快照
    this.snapshots[this.currentIndex] = snapshot;
    this.currentIndex = (this.currentIndex + 1) % this.maxSnapshots;
  }
  
  async rollbackToFrame(targetFrame) {
    // 找到目标帧的快照
    const snapshot = this.snapshots.find(s => s.frame === targetFrame);
    if (!snapshot) throw new Error('Snapshot not found');
    
    // 验证完整性
    if (!verifyChecksum(snapshot.state, snapshot.checksum)) {
      throw new Error('Checksum verification failed');
    }
    
    return deepClone(snapshot.state);
  }
}
```

### 5.2 状态管理 (Pinia)

#### 5.2.1 Store 模块化

```typescript
// stores/game.ts
export const useGameStore = defineStore('game', () => {
  // State
  const currentPlayer = ref<number>(0)
  const phase = ref<'betting' | 'playing' | 'settling'>('betting')
  const tiles = ref<string[]>([])
  const discardPile = ref<string[]>([])
  
  // Getters
  const isPlayerTurn = computed(() => (playerId: number) => {
    return currentPlayer.value === playerId
  })
  
  // Actions
  function drawTile(tile: string) {
    tiles.value.push(tile)
  }
  
  function discardTile(index: number) {
    discardPile.value.push(tiles.value.splice(index, 1)[0])
  }
  
  return {
    currentPlayer,
    phase,
    tiles,
    discardPile,
    isPlayerTurn,
    drawTile,
    discardTile
  }
})
```

```javascript
class ObjectPool {
  constructor(createFn, resetFn, initialSize = 100) {
    this.createFn = createFn;
    this.resetFn = resetFn;
    this.pool = [];
    
    // 预分配对象
    for (let i = 0; i < initialSize; i++) {
      this.pool.push(createFn());

  }
  
  release(obj) {

  }
}
```

```javascript
// 四叉树实现 AOI (Area Of Interest)
class QuadTree {
  constructor(bounds, capacity = 4) {
    this.bounds = bounds;
    this.capacity = capacity;
    this.objects = [];
    this.divided = false;
  }
  
  insert(obj) {
    if (!this.bounds.contains(obj.position)) return false;
    
    if (this.objects.length < this.capacity && !this.divided) {
      this.objects.push(obj);
      return true;
    }
    
    if (!this.divided) this.subdivide();
    
    return this.northeast.insert(obj) ||
           this.northwest.insert(obj) ||
           this.southeast.insert(obj) ||
           this.southwest.insert(obj);
  }
  
  query(range) {
    const found = [];
    
    if (!this.bounds.intersects(range)) return found;
    
    for (const obj of this.objects) {
      if (range.contains(obj.position)) {
        found.push(obj);
      }
    }
    
    if (this.divided) {
      found.push(...this.northeast.query(range));
      found.push(...this.northwest.query(range));
      found.push(...this.southeast.query(range));
      found.push(...this.southwest.query(range));
    }
    
    return found;
  }
}
```

```javascript
class MultiLevelCache {
  constructor() {
    this.l1 = new Map();  // 本地缓存
    this.l2 = redis;      // Redis 缓存
    this.ttl = {
      l1: 60 * 1000,      // 1 分钟
      l2: 5 * 60 * 1000   // 5 分钟
    };
  }
  
  async get(key) {
    // 1. 尝试 L1
    const l1Value = this.l1.get(key);
    if (l1Value) return l1Value;
    
    // 2. 尝试 L2
    const l2Value = await this.l2.get(key);
    if (l2Value) {
      // 回填 L1
      this.l1.set(key, l2Value);
      setTimeout(() => this.l1.delete(key), this.ttl.l1);
      return l2Value;
    }
    
    return null;
  }
  
  async set(key, value) {
    // 同时写入 L1 和 L2
    this.l1.set(key, value);
    setTimeout(() => this.l1.delete(key), this.ttl.l1);
    
    await this.l2.setex(key, this.ttl.l2 / 1000, JSON.stringify(value));
  }
}
```

```javascript
// 布隆过滤器
class BloomFilter {
  constructor(size = 1000000, hashCount = 3) {
    this.size = size;
    this.hashCount = hashCount;
    this.bits = new Array(size).fill(false);
  }
  
  add(item) {
    for (let i = 0; i < this.hashCount; i++) {
      const index = this.hash(item, i) % this.size;
      this.bits[index] = true;
    }
  }
  
  mightContain(item) {
    for (let i = 0; i < this.hashCount; i++) {
      const index = this.hash(item, i) % this.size;
      if (!this.bits[index]) return false;
    }
    return true;
  }
}

// 使用示例

  // 1. 布隆过滤器检查

  if (room) return room;
```

### 10.5 悔棋与容错

#### 10.5.1 悔棋功能

```javascript
class RegretSystem {
  constructor() {
    this.moveStack = [];
    this.maxRegretSteps = 1;
  }
  
  recordMove(move) {
    this.moveStack.push({
      ...move,
      timestamp: Date.now()
    });
    
    // 限制栈大小
    if (this.moveStack.length > this.maxRegretSteps) {
      this.moveStack.shift();
    }
  }
  
  async requestRegret(playerId, roomId) {
    // 联机模式需要对方同意
    const room = await GameStore.getRoom(roomId);
    const opponent = room.getOpponent(playerId);
    
    // 发送悔棋请求
    socket.emit('regretRequest', {
      roomId,
      requester: playerId,
      target: opponent.id
    });
    
    // 等待对方响应
    return new Promise((resolve) => {
      socket.once('regretResponse', (response) => {
        if (response.approved) {
          this.rollback();
          resolve({ success: true });
        } else {
          resolve({ success: false, reason: 'OPPONENT_REFUSED' });
        }
      });
      
      // 30 秒超时
      setTimeout(() => {
        resolve({ success: false, reason: 'TIMEOUT' });
      }, 30000);
    });
  }
  
  rollback() {
    const lastMove = this.moveStack.pop();
    // 恢复游戏状态
    return lastMove;
  }
}
```
